let frames = []; let maxFrames = 80; function setup() { createCanvas(windowWidth, windowHeight); initFloatingSystem(); } function initFloatingSystem() { frames = []; for (let i = 0; i < 50; i++) { let tamaño = random(50, 95); let rarezas = ["common", "rare", "epic", "legendary"]; let r = random() > 0.85 ? random(rarezas.slice(1)) : "common"; frames.push({ pfp: null, rarity: r, score: rarityScore(r), x: random(0, width - tamaño), y: random(0, height - tamaño), size: tamaño, vx: random(-0.2, 0.2), vy: random(-0.2, 0.2), t: random(1000), col: color(random(100, 255), random(100, 255), random(220, 255)) }); } } function rarityScore(r) { return { common: 1, rare: 2, epic: 3, legendary: 4 }[r] || 1; } function draw() { // Nota: si te da error porque no tenés la imagen subida al editor, // podés cambiar provisoriamente la línea de abajo por: background(20); background(20); for (let f of frames) { drawAndMoveFrame(f); } } function drawAndMoveFrame(f) { f.x += f.vx; f.y += f.vy; if (f.x < 0 || f.x > width - f.size) f.vx *= -1; if (f.y < 0 || f.y > height - f.size) f.vy *= -1; push(); translate(f.x, f.y); let glow = f.score * 40; let pulse = sin(frameCount * 0.02 + f.t) * 15; noFill(); stroke(f.pfp ? 255 : f.col, 110 + glow); strokeWeight(f.score); rect(0, 0, f.size, f.size); if (f.pfp) { image(f.pfp, 2, 2, f.size - 4, f.size - 4); } else { fill(red(f.col), green(f.col), blue(f.col), 20 + pulse); rect(2, 2, f.size - 4, f.size - 4); fill(255, 90 + pulse); noStroke(); if (f.rarity === "legendary") { ellipse(f.size/2, f.size/2, f.size * 0.35); } else if (f.rarity === "epic") { rect(f.size * 0.35, f.size * 0.35, f.size * 0.3, f.size * 0.3); } } pop(); } function addPFP(url, rarity = "common") { loadImage(url, (img) => { if (frames.length >= maxFrames) { frames.shift(); } let tamaño = random(80, 130); frames.push({ pfp: img, rarity: rarity, score: rarityScore(rarity), x: random(0, width - tamaño), y: random(0, height - tamaño), size: tamaño, vx: random(-0.25, 0.25), vy: random(-0.25, 0.25), t: random(1000) }); }); } function windowResized() { resizeCanvas(windowWidth, windowHeight); }
